156   /**
157    * Record a sale of the named asset (stock or cryptocurrency) at the current live market value if
158    * we hold that asset. The sale price should be the real live price of the asset at the time of
159    * sale retrieved from an appropriate web API. The revenue generated from the sale should be added
160    * to the total funds available to the user.
161    *
162    * <p>Business logic: If we hold > 1 units of the specified asset (say 10 units of Microsoft stock
163    * MSFT), and the parameter amount is less than total units of the stock, we should sell the
164    * units that maximise our profit. Remember some of the stock could have been purchased on
165    * different dates and therefore have been purchased at different price points.
166    *
167    * @param assetSymbol the name of the asset (stock symbol or cryptocurrency) to sell.
168    * @param amount    the amount of the asset to sell.
169    * @return True if the asset is sold successfully, otherwise false (we may not have that asset in
170    *       our portfolio).
171    */
172   @Override
173   public boolean sellAsset(String assetSymbol, double amount) {
174     double amountLeftToSell = amount;
175 
176     //Checks if asset symbol is valid
177     if (!checkAssetSymbol(assetSymbol)) {
178       return false;
179     }
180 
181     // 1. Get all the holdings with this assetSymbol.
182     List<Asset> assetsHoldings = getAssetsWithHoldings(assetSymbol);
183     if (assetsHoldings == null) {
184       return false;
185     }
186 
187     // 2. Check if there is sufficient balance across all of them.
188     double currentUnitsBalance = assetsHoldings.stream()
189         .mapToDouble(Asset::getCurrentUnitsBalance).sum();
190 
191     if (currentUnitsBalance < amountLeftToSell) {
192       // we don't own enough of the asset to sell.
193       return false;
194     }
195 
196     // 3. Get the live price for this asset
197     AssetQuote assetQuote = getAssetQuote(assetSymbol);
198     //If no live price is available, we cannot sell the asset.
199     if (assetQuote == null) {
200       return false;
201     }
202 
203     // 3. Work out which holdings will yield the best profit.
204     // There are a few ways to do this. One such way is to sort the list by the purchase price.
205     assetsHoldings.sort(compareByPurchasePrice);
206 
207     // 4. We will now iterate through the holdings to sell, up to the amount required.
208     for (Asset asset : assetsHoldings) {
209       // How much of this asset do we need to sell
210       double assetUnitsBalance = asset.getCurrentUnitsBalance();
211 
212       if (assetUnitsBalance >= amountLeftToSell) {
213         // This holding has enough units to fulfill the sell.
214         asset.sellHolding(amountLeftToSell, assetQuote.getRegularMarketPrice());
215         amountLeftToSell = 0;
216       } else {
217         asset.sellHolding(assetUnitsBalance, assetQuote.getRegularMarketPrice());
218         amountLeftToSell -= assetUnitsBalance;
219       }
220 
221       if (amountLeftToSell == 0) {
222         // all sold.
223         break;
224       }
225     }
226 
227     // 5. The revenue generated from the sale should be added to the total funds
228     // available to the user. We know how many units we have sold, and we know the live price.
229     double revenueGenerated = amount * assetQuote.getRegularMarketPrice();
230     availableFunds += revenueGenerated;
231     return true;
232   }
